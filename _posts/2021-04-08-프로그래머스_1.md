# 3진법


```python
def solution(n):
    answer = 0
    y = 0
    for i in range(17,-1,-1):
        if n//(3**i) >= 1: 
            y = i
            break
        
    thr = np.zeros(shape=(y+1,),dtype=np.int8) 
    
    for j in range(y, -1, -1):
        if n//(3**j) >=1 : 
            thr[j] = n//(3**j)
            n = n - (n//(3**j))*(3**j)
        elif n//(3**j) == 0 :
            thr[j] = 0
            
    for k in range(len(thr)) :
        answer += thr[k]*(3**(len(thr)-1-k))
    return answer
    
```
